165  template <typename Func, typename... Args>
166  unsigned int connectMenuItem(const std::vector<String>& hierarchy, Func&& handler, const Args&... args)
167  {
168 #if defined(__cpp_lib_invoke) && (__cpp_lib_invoke >= 201411L)
169  return onMenuItemClick.connect(
170  [=](const std::vector<String>& clickedMenuItem)
171  {
172  if (clickedMenuItem == hierarchy)
173  std::invoke(handler, args...);
174  }
175  );
176 #else
177  return onMenuItemClick.connect(
178  [f=std::function<void(const Args&...)>(handler),args...,hierarchy](const std::vector<String>& clickedMenuItem)
179  {
180  if (clickedMenuItem == hierarchy)
181  f(args...);
182  }
183  );
184 #endif
185  }
