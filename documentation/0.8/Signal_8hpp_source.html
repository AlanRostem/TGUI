393  #define TGUI_SIGNAL_VALUE_DECLARATION(TypeName, Type) \
394  /**************************************************************************************************************************** \
395  * @brief Signal to which the user can subscribe to get callbacks from
396  ****************************************************************************************************************************/ \
397  class TGUI_API Signal##TypeName : public Signal \
398  { \
399  public: \
400  using Delegate##TypeName = std::function<void(Type)>; \
401  using Delegate##TypeName##Ex = std::function<void(std::shared_ptr<Widget>, const std::string&, Type)>; \
402  using Signal::connect; \
403  \
404  /************************************************************************************************************************ \
405  * @brief Constructor
406  ************************************************************************************************************************/ \
407  Signal##TypeName(std::string&& name) : \
408  Signal{std::move(name), 1} \
409  { \
410  } \
411  \
412  /************************************************************************************************************************ \
413  * @brief Connects a signal handler that will be called when this signal is emitted
414  *
415  * @param handler Callback function that is given a Vector2f as argument
416  *
417  * @return Unique id of the connection
418  ************************************************************************************************************************/ \
419  unsigned int connect(const Delegate##TypeName& handler); \
420  \
421  /************************************************************************************************************************ \
422  * @brief Connects a signal handler that will be called when this signal is emitted
423  *
424  * @param handler Callback function that is given a pointer to the widget, the name of the signal and a Vector2f as arguments
425  *
426  * @return Unique id of the connection
427  ************************************************************************************************************************/ \
428  unsigned int connect(const Delegate##TypeName##Ex& handler); \
429  \
430  /************************************************************************************************************************ \
431  * @internal
432  * @brief Call all connected signal handlers
433  ************************************************************************************************************************/ \
434  bool emit(const Widget* widget, Type param) \
435  { \
436  if (m_handlers.empty()) \
437  return false; \
438  \
439  internal_signal::parameters[1] = static_cast<const void*>(&param); \
440  return Signal::emit(widget); \
441  } \
442  \
443  private: \
444  /************************************************************************************************************************ \
445  * @internal
446  * @brief Checks whether the unbound parameters match with this signal
447  * @return The index in the parameters list where the parameters will be stored
448  ************************************************************************************************************************/ \
449  unsigned int validateTypes(std::initializer_list<std::type_index> unboundParameters) const override; \
450  };
